** Installation

*** lazy.nvim

    #+begin_src lua
    {
      "chipsenkbeil/org-roam.nvim",
      dependencies = { "nvim-orgmode/orgmode" },
      config = function()
        require("org-roam").setup({
          org_roam_directory = "~/orgfiles",
        })
      end
    }
    #+end_src

*** packer.nvim

    #+begin_src lua
    use {
      "chipsenkbeil/org-roam.nvim",
      requires = { "nvim-orgmode/orgmode" },
      config = function()
        require("org-roam"):setup({
          org_roam_directory = "~/orgfiles",
        })
      end
    }
    #+end_src
